```python
from notifier.gate_notifications import notify_l1_result, notify_l2_result
from notifier.daily_summary import get_daily_summary

# L1 快速门控后
async def l1_quick_review(...):
    result = await self._do_l1_review(...)
    
    # 发送通知
    notify_l1_result(
        signal_id=signal_id,
        strategy_id=strategy_id,
        model_id=self.MODEL,
        symbol=symbol,
        signal_direction=signal,
        passed=result["pass"],
        risk_flag=result["risk_flag"],
        confidence=result["confidence"],
        reasoning=result["reasoning"],
        trace_id=trace_id,
        degraded=result.get("degraded", False),
    )
    
    # 记录到每日汇总
    get_daily_summary().record_l1(symbol, result["pass"], result["risk_flag"])
    
    return result

# L2 深审同理
```

### 3. 配置定时任务

```python
import schedule
from notifier.daily_summary import get_daily_summary, send_daily_summary
from notifier.health_monitor import get_health_monitor

# 每日 23:55 发送门控日报
schedule.every().day.at("23:55").do(lambda: send_daily_summary(get_daily_summary()))

# 每 15 分钟健康快照
def health_snapshot():
    monitor = get_health_monitor()
    snapshot = monitor.take_snapshot()
    monitor.send_snapshot_notification(snapshot)

schedule.every(15).minutes.do(health_snapshot)

# 每日 08:00 发送健康报告
schedule.every().day.at("08:00").do(lambda: get_health_monitor().send_daily_report())
```
