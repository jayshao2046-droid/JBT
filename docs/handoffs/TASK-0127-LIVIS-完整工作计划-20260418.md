        self, 
        yaml_path: str, 
        start_date: str,

    ) -> str:
        """提交回测任务，返回 task_id"""
        # 1. 读取 YAML

        # 3. POST /api/backtest/run
        # 4. 返回 task_id
    
    async def poll_result(
        self, 
        task_id: str, 
        timeout: int = 600
    ) -> dict:
        """轮询获取回测结果"""
        # 1. 每 10 秒查询一次
        # 2. GET /api/backtest/results/{task_id}
        # 3. 状态为 completed 时返回结果
        # 4. 超时或失败时抛出异常
    
    def validate_yaml(self, yaml_path: str) -> tuple[bool, list[str]]:
        """验证 YAML 是否符合 TqSdk 标准"""
        # 检查必填字段
        # 检查禁止字段
        # 返回 (是否通过, 错误列表)
    
    def convert_to_main_contract(self, symbol: str) -> str:
        """转换为主力合约代码"""
        # DCE.p0 → DCE.p2505

### 阶段 2：开发主控脚本（预计 2 小时）

#### 2.1 主控脚本架构
**文件**：`services/decision/scripts/run_full_pipeline_35_symbols.py`

**核心流程**：
```python
async def main():
    # 1. 初始化所有组件
    generator = LLMStrategyGenerator()
    optimizer = StrategyParamOptimizer()
    evaluator = StrategyEvaluator()
    local_executor = YAMLSignalExecutor()
    tqsdk_client = TqSdkBacktestClient()
    evidence_mgr = EvidenceManager()
    feishu_notifier = FeishuStrategyNotifier()
    
    # 2. 35 品种串行循环
    for symbol in SYMBOLS_35:
        logger.info(f"开始处理品种: {symbol}")
        
        # 2.1 生成策略矩阵（7-10 个策略）
        strategies = await generator.generate_strategies(symbol)
        
        # 2.2 逐个策略处理
        for strategy in strategies:
            # a. 调优
            opt_result = await optimizer.optimize(strategy)
            
            # b. 本地回测
            local_result = await local_executor.execute(strategy)
            
            # c. TqSdk 回测
            task_id = await tqsdk_client.submit_backtest(strategy)
            tqsdk_result = await tqsdk_client.poll_result(task_id)
            
            # d. 评分
            eval_result = await evaluator.evaluate_strategy(strategy)
            
            # e. 证据留存
            evidence_mgr.save_all_reports(strategy, ...)
            storage_path = evidence_mgr.move_to_bucket(
                strategy, symbol, eval_result['score']
            )
            
            # f. 飞书通知
            await feishu_notifier.notify_strategy_completed(
                symbol=symbol,
                strategy_name=strategy['name'],
                category=strategy['category'],
                factors=strategy['factors'],
                params=strategy['params'],
                local_result=local_result,
                tqsdk_result=tqsdk_result,
                final_score=eval_result['score'],
                grade=eval_result['grade'],
                optimization_trials=opt_result['n_trials'],
                storage_path=storage_path
            )
        
        # 2.3 输出品种汇总
        logger.info(f"品种 {symbol} 完成")
    
    # 3. 输出全局汇总
    logger.info("35 品种全部完成")
```

## 六、风险与应对

### 6.1 技术风险
| 风险 | 影响 | 应对措施 |
|------|------|---------|
| TqSdk 回测超时 | 流程阻塞 | 增加超时重试，最多 3 次 |
| Mini data API 不稳定 | 本地回测失败 | 增加重试机制，缓存历史数据 |
| 飞书 Webhook 限流 | 通知丢失 | 增加发送间隔，失败重试 |
| 主力合约查询失败 | 合约代码错误 | 降级到固定月份映射表 |

### 6.2 时间风险
| 风险 | 影响 | 应对措施 |
|------|------|---------|
| 单品种耗时超预期 | 总时长超 30 小时 | 优化调优 trials 数量（100 → 50） |
| TqSdk 回测排队 | 等待时间过长 | 错峰执行，避开高峰期 |

### 6.3 质量风险
| 风险 | 影响 | 应对措施 |
|------|------|---------|
| 双回测结果差异过大 | 策略可信度低 | 增加差异检查，超过阈值告警 |
| 评分逻辑错误 | 错误分桶 | 增加评分验证，人工抽查 |
