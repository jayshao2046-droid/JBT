### 3a. kline_analyzer.py — `fetch_kline()` 改为 tushare 日 K

**现状**：
```python
async def fetch_kline(self, symbol: str, duration: str = "1min", count: int = 120):
    url = f"{self.data_api}/api/v1/bars"   # Mini API → 404
```

**目标**：用 tushare `fut_daily` 拉取主力连续合约日 K，改为同步函数（tushare SDK 是同步的）：

```python
def fetch_daily_kline(self, symbol: str, count: int = 60) -> Optional[List[Dict]]:
    """
    从 tushare 拉取日 K 数据（主力连续合约）

    Args:
        symbol: KQ.m@SHFE.rb 格式
        count: 拉取条数（默认60，约3个月）

    Returns:
        K线列表 [{datetime, open, high, low, close, volume, oi}, ...]
    """
    import tushare as ts
    from datetime import datetime, timedelta

    token = ResearcherConfig.TUSHARE_TOKEN
    if not token:
        logger.warning("[DAILY-KLINE] TUSHARE_TOKEN 未配置")
        return None

    ts_code = _convert_to_tushare_code(symbol)  # KQ.m@SHFE.rb → RB9999.SHF
    if not ts_code:
        return None

    try:
        pro = ts.pro_api(token)
        end_date = datetime.now().strftime("%Y%m%d")
        start_date = (datetime.now() - timedelta(days=count * 2)).strftime("%Y%m%d")
        df = pro.fut_daily(ts_code=ts_code, start_date=start_date, end_date=end_date)
        import time; time.sleep(0.3)   # rate limit
        if df is None or df.empty:
            logger.warning(f"[DAILY-KLINE] {ts_code} tushare 返回空")
            return None
        df = df.sort_values("trade_date").tail(count)
        bars = []
        for _, row in df.iterrows():
            bars.append({
                "datetime": str(row["trade_date"]),
                "open": float(row.get("open", 0)),
                "high": float(row.get("high", 0)),
                "low": float(row.get("low", 0)),
                "close": float(row.get("close", 0)),
                "volume": float(row.get("vol", 0)),
                "open_oi": float(row.get("oi", 0)),
            })
        logger.info(f"[DAILY-KLINE] {ts_code} 获取 {len(bars)} 条日K")
        return bars
    except Exception as e:
        logger.warning(f"[DAILY-KLINE] {ts_code} tushare 拉取失败: {e}")
        return None
```

**symbol 转换函数**（新增，放在模块级）：

```python
def _convert_to_tushare_code(symbol: str) -> Optional[str]:
    """
    KQ.m@SHFE.rb → RB9999.SHF （主力连续合约，tushare 用 9999 结尾）
    KQ.m@DCE.i   → I9999.DCE
    KQ.m@CZCE.TA → TA9999.ZCE
    """
    EXCHANGE_MAP = {
        "SHFE": "SHF",
        "DCE": "DCE",
        "CZCE": "ZCE",
        "CFFEX": "CFX",
        "INE": "INE",
        "GFEX": "GFE",
    }
    try:
        # 格式: KQ.m@SHFE.rb
        exchange_part, code_part = symbol.split("@")[-1].split(".")
        exchange = EXCHANGE_MAP.get(exchange_part.upper(), exchange_part)
        ts_code = f"{code_part.upper()}9999.{exchange}"
        return ts_code
    except Exception:
        return None
```

**同步化 `analyze_symbol()`**：由于 `fetch_daily_kline()` 是同步的，需要在异步方法中用 `asyncio.get_event_loop().run_in_executor()` 包装，或直接改为在同步上下文中调用：

### 3d. decision pipeline.py — `_fetch_kline_data()` 改为 tqsdk 主路

**现状**：
```python
async def _fetch_kline_data(self, data_service_url, symbol, timeframe, ...):
    url = f"{data_service_url}/api/v1/bars"   # Mini API → 404
    response = await client.get(url, params=params)
    return data.get("bars", [])
```

**目标**：tqsdk 作为主路，Mini API 降为 fallback：

```python
async def _fetch_kline_data(
    self,
    data_service_url: str,
    symbol: str,
    timeframe: str,
    start_date: Optional[str] = None,
    end_date: Optional[str] = None,
) -> list:
    """从 tqsdk 拉取实时分钟 K 线，Mini API 降为 fallback"""
    # 1. 尝试 tqsdk
    tqsdk_bars = await self._fetch_kline_via_tqsdk(symbol, timeframe)
    if tqsdk_bars:
        return tqsdk_bars
    # 2. fallback: Mini API
    try:
        url = f"{data_service_url.rstrip('/')}/api/v1/bars"
        params = {"symbol": symbol, "timeframe": timeframe}
        if start_date:
            params["start"] = start_date
        if end_date:
            params["end"] = end_date
        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.get(url, params=params)
            response.raise_for_status()
            return response.json().get("bars", [])
    except Exception as e:
        logger.warning(f"[KLINE] Mini API fallback 也失败: {e}")
        return []

async def _fetch_kline_via_tqsdk(self, symbol: str, timeframe: str) -> list:
    """
    通过 tqsdk 拉取实时分钟 K 线
    symbol 格式: KQ.m@SHFE.rb (tqsdk 原生)
    timeframe: "1min" / "5min" / "15min"
    """
    import os
    username = os.getenv("TQSDK_AUTH_USERNAME", "")
    password = os.getenv("TQSDK_AUTH_PASSWORD", "")
    if not username or not password:
        logger.warning("[KLINE] TQSDK 账号未配置，跳过")
        return []

    duration_map = {"1min": 60, "5min": 300, "15min": 900, "1d": 86400}
    duration_seconds = duration_map.get(timeframe, 60)

    try:
        import asyncio
        loop = asyncio.get_event_loop()
        bars = await loop.run_in_executor(
            None,
            self._sync_fetch_tqsdk,
            symbol, duration_seconds, username, password
        )
        return bars
    except Exception as e:
        logger.warning(f"[KLINE] tqsdk 拉取失败: {e}")
        return []

@staticmethod
def _sync_fetch_tqsdk(symbol: str, duration_seconds: int, username: str, password: str) -> list:
    """同步 tqsdk 拉取（在 executor 中运行，避免阻塞事件循环）"""
    from tqsdk import TqApi, TqAuth
    api = None
    try:
        api = TqApi(auth=TqAuth(username, password))
        klines = api.get_kline_serial(symbol, duration_seconds, data_length=120)
        api.wait_update()  # 等待数据就绪
        bars = []
        for _, row in klines.iterrows():
            if row.get("close", 0) > 0:
                bars.append({
                    "datetime": str(row.get("datetime", "")),
                    "open": float(row.get("open", 0)),
                    "high": float(row.get("high", 0)),
                    "low": float(row.get("low", 0)),
                    "close": float(row.get("close", 0)),
                    "volume": float(row.get("volume", 0)),
                    "open_oi": float(row.get("open_oi", 0)),
                })
        return bars
    except Exception as e:
        logger.warning(f"[KLINE] tqsdk sync fetch 失败: {e}")
        return []
    finally:
        if api:
            try:
                api.close()
            except Exception:
                pass
```

## 7. 注意事项

1. **tushare 9999 vs 0000**：`9999` 是主力连续合约（价格连续），`0000` 也可以；两者在 tushare 中均可用。优先用 `9999`。
2. **tqsdk 在 decision 容器中**：decision 服务通过 backtest 调用 tqsdk，但 `pipeline.py` 本身不直接 import tqsdk。需确认 decision 容器内已安装 tqsdk（`pip show tqsdk`）。如未安装：`docker exec JBT-DECISION-8104 pip install tqsdk`
3. **tqsdk 并发**：每个 TqApi 实例连接天勤服务器，L1/L2 频繁调用时不要复用实例，用完即 `api.close()`。
4. **Alienware tushare 安装**：已确认 TASK-0117-004 中容器已安装 tushare。
5. **非交易日**：盘后 16:00 触发需加交易日判断，非交易日跳过。
