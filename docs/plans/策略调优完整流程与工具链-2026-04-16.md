请生成以下因子的 Python 代码：

2. oi_change_rate: 持仓量变化率
3. price_ma_distance: 价格与 20 日均线距离

- 使用 pandas 和 numpy
- 处理缺失值
- 标准化输出（z-score）
- 添加单元测试

# DeepSeek Coder 输出
deepseek_output = """
import pandas as pd
import numpy as np

def calculate_volume_ma_ratio(df: pd.DataFrame, window: int = 20) -> pd.Series:
    '''成交量 / 20 日均量'''
    volume_ma = df['volume'].rolling(window=window).mean()
    ratio = df['volume'] / volume_ma
    return (ratio - ratio.mean()) / ratio.std()  # z-score 标准化

def calculate_oi_change_rate(df: pd.DataFrame) -> pd.Series:
    '''持仓量变化率'''
    oi_change = df['open_interest'].pct_change()
    return (oi_change - oi_change.mean()) / oi_change.std()

def calculate_price_ma_distance(df: pd.DataFrame, window: int = 20) -> pd.Series:
    '''价格与 20 日均线距离'''
    price_ma = df['close'].rolling(window=window).mean()
    distance = (df['close'] - price_ma) / price_ma
    return (distance - distance.mean()) / distance.std()

# 单元测试
def test_factors():
    df = pd.DataFrame({
        'close': [100, 102, 101, 103, 105],
        'volume': [1000, 1200, 1100, 1300, 1400],
        'open_interest': [5000, 5100, 5050, 5200, 5300],
    })
    
    assert len(calculate_volume_ma_ratio(df)) == len(df)
    assert len(calculate_oi_change_rate(df)) == len(df)
    assert len(calculate_price_ma_distance(df)) == len(df)

#### 步骤 2：套利策略生成（DeepSeek Coder）

```python
# 2. DeepSeek Coder 生成套利策略代码
deepseek_prompt = f"""
根据 phi4 的价差分析结果：
{phi4_output}

请生成棕榈油-豆油套利策略代码：
1. 监控价差（p - y）
2. 当价差 > 均值 + 2*标准差时，做空价差（做空 p，做多 y）
3. 当价差 < 均值 - 2*标准差时，做多价差（做多 p，做空 y）
4. 当价差回归均值时，平仓

要求：
- 使用 pandas 和 numpy
- 计算协整关系
- 动态调整仓位（根据价差偏离程度）
- 添加风控（最大回撤、止损）
"""

# DeepSeek Coder 输出
deepseek_output = """
import pandas as pd
import numpy as np
from statsmodels.tsa.stattools import coint

class SpreadArbitrageStrategy:
    def __init__(self, symbol1='p', symbol2='y', window=60):
        self.symbol1 = symbol1
        self.symbol2 = symbol2
        self.window = window
    
    def calculate_spread(self, df1: pd.DataFrame, df2: pd.DataFrame) -> pd.Series:
        '''计算价差'''
        return df1['close'] - df2['close']
    
    def check_cointegration(self, spread: pd.Series) -> bool:
        '''检查协整关系'''
        _, pvalue, _ = coint(spread, np.arange(len(spread)))
        return pvalue < 0.05
    
    def generate_signals(self, spread: pd.Series) -> pd.Series:
        '''生成交易信号'''
        mean = spread.rolling(window=self.window).mean()
        std = spread.rolling(window=self.window).std()
        
        z_score = (spread - mean) / std
        
        signals = pd.Series(0, index=spread.index)
        signals[z_score > 2] = -1  # 做空价差
        signals[z_score < -2] = 1   # 做多价差
        signals[abs(z_score) < 0.5] = 0  # 平仓
        
        return signals
    
    def calculate_position_size(self, z_score: float) -> float:
        '''动态调整仓位'''
        return min(abs(z_score) / 2, 1.0)  # 最大 100% 仓位
"""
```
